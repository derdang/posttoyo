character_event = {
	id = cetic.10
	desc = EVTDESC_bounce
	picture = GFX_evt_chinese_bureaucrat
	is_triggered_only = yes
	hide_window = yes
	only_playable = yes
	
	trigger = {
		OR = {
			primary_title = {
				government = sinic_bureaucracy_government
			}
		}
	}
	
	immediate = {
		current_heir = { 
			add_character_modifier = {
				name = imperial_authority
				duration = -1
			}
			if = {
				limit = {
					check_variable = { which = global_imp_auth value < 10 }
				}
				add_trait = weak_emperor
			}
			if = {
				limit = {
					check_variable = { which = global_imp_auth value < 50 }
				}
				add_trait = figurehead
				character_event = { id = cetic.316 }
			}
			if = {
				limit = {
					AND = {
						check_variable = { which = global_imp_auth value = 50 }
						check_variable = { which = global_imp_auth value < 70 }
					}
				}
				set_character_flag = imp_auth_dont_change
				character_event = { id = cetic.316 }
			}
			if = {
				limit = {
					check_variable = { which = global_imp_auth value > 70 }
				}
				set_variable = { which = global_imp_auth value = 55 }
				set_character_flag = imp_auth_dont_change
			}
		}
	}
	
	option = { name = OK }
}

#If the emperor somehow loses the throne
character_event = {
	id = cetic.363
	hide_window = yes
	only_playable = yes
	
	trigger = {
		AND = {
			government = sinic_bureaucracy_government
			independent = yes
		}
		NOT = {
			has_character_modifier = imperial_authority
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	immediate = {
		any_character = {
			limit = {
				AND = {
					government = sinic_bureaucracy_government
					independent = yes
				}
			}
			remove_trait = weak_emperor
			remove_trait = strong_emperor
			remove_trait = figurehead
			remove_character_modifier = imperial_authority
			clr_character_flag = era_name
		}
		add_character_modifier = {
			name = imperial_authority
			duration = -1
		}
		if = {
			limit = {
				check_variable = { which = global_imp_auth value < 10 }
				NOT = {
					trait = weak_emperor
				}
			}
			add_trait = weak_emperor
			remove_trait = figurehead
		}
		if = {
			limit = {
				check_variable = { which = global_imp_auth value >= 10 }
				trait = weak_emperor
			}
			remove_trait = weak_emperor
		}
		if = {
			limit = {
				check_variable = { which = global_imp_auth value >= 10 }
				check_variable = { which = global_imp_auth value < 50 }
				NOT = {
					trait = figurehead
				}
			}
			add_trait = figurehead
		}
		if = {
			limit = {
				check_variable = { which = global_imp_auth value > 50 }
				trait = figurehead
			}
			remove_trait = figurehead
		}
		if = {
			limit = {
				check_variable = { which = global_imp_auth value < 70 }
				trait = strong_emperor
			}
			remove_trait = strong_emperor
		}
		if = {
			limit = {
				check_variable = { which = global_imp_auth value > 70 }
				NOT = {
					trait = strong_emperor
				}
			}
			add_trait = strong_emperor
		}
	}
}
