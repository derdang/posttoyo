		trait = sympathy_zoroastrianism
	}
	trigger_else_if = {
		limit = { PREV = { true_religion_group = jewish_group } }
		trait = sympathy_judaism
	}
	trigger_else_if = {
		PREV = { true_religion_group = indian_group }
		trait = sympathy_indian
	}
	trigger_else_if = {
		PREV = { true_religion_group = buddhist_group }
		trait = sympathy_buddhist
	}
	trigger_else_if = {
		PREV = { true_religion_group = japonic }
		trait = sympathy_japonic
	}
	trigger_else_if = {
		PREV = { true_religion_group = eastern }
		trait = sympathy_eastern
	}
	trigger_else_if = {
		PREV = { true_religion_group = latter_day_saints }

is_in_prevs_interesting_society_trigger = {
	# Christian
	trigger_if = {
		limit = { society_member_of = secret_religious_society_catholic }
		PREV = { interested_in_society = secret_religious_society_catholic }
	}
	trigger_else_if = {
		limit = { society_member_of = secret_religious_society_restorationist }
		PREV = { interested_in_society = secret_religious_society_restorationist }
	}
	# Latter Day Saints
	trigger_else = {
		society_member_of = secret_religious_society_mormon
		PREV = { interested_in_society = secret_religious_society_mormon }
	}
}

# Remember that Nomads can safely change religion
# Tooltips only shows anything if player is missing required DLC
has_safe_religion = {
	show_only_failed_conditions = yes

	trigger_if = {
		limit = { religion_group = jewish_group }
		has_dlc = "Sons of Abraham"
	}
	trigger_else_if = {
		limit = { religion_group = muslim }
		has_dlc = "The Sword of Islam"
	}
	trigger_else_if = {
		limit = { religion_group = pagan_group }

		OR = {
			has_dlc = "The Old Gods"
			has_dlc = "Holy Fury"
		}
	}
	trigger_else_if = {
		limit = { OR = {religion_group = zoroastrian_group religion_group = oldworld } }
		has_dlc = "The Old Gods"
	}
	trigger_else_if = {
		limit = { religion_group = eastern }

		OR = {
			has_dlc = "Rajas of India"
			has_dlc = "Jade Dragon"
		}
	}
	trigger_else_if = {
		limit = { OR = {religion_group = indian_group religion_group = buddhist_group } }
		has_dlc = "Rajas of India"
	}
}
