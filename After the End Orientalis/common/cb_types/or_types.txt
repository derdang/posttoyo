	can_use = {

			any_realm_title = {
				location = {
					region = custom_korea

			}
		}
	}

			OR = {
				holder_scope = {
					OR = {

			limit = { NOT = { character = ROOT } }

				participation_scaled_prestige = 50

							has_ambition = obj_strengthen_religion
						}
						change_variable = { which = strengthen_religion value = 1 }
					}
				}
			}
		}
		
		ROOT = { 
			participation_scaled_piety = 100
			participation_scaled_prestige = 50
			religion_authority = {
				modifier = won_holy_war
				years = 20

					limit = {

					years = 20

		FROM = { prestige = -50 }
	}

		}
		any_defender = {

			hidden_tooltip = {

				participation_scaled_prestige = 100

							has_ambition = obj_strengthen_religion

				years = 20
			}
		}
		FROM = { 
			religion_authority = {
				modifier = won_holy_war
				years = 20
			}
			hidden_tooltip = {
				if = {
					limit = {
						has_ambition = obj_strengthen_religion
					}
					change_variable = { which = strengthen_religion value = 1 }

			FROM = {
				is_nomadic = yes
			}
			ROOT = {
				is_nomadic = no
				NOT = {
					is_tribal = yes
				}
			}
			NOT = {
				any_direct_de_jure_vassal_title = {
					any_direct_de_jure_vassal_title = {
						OR = {
							holding_type = castle
							holding_type = city
